Bootstrap-wysihtml5 demo

Overview

This page shows how the spellchecker can be integrated into Bootstrap-wysihtml5.

Usage

(function() {
      
  var element = $('.textarea');
  
  element.wysihtml5({
    toolbar: {
       "spellchecker":
          "<li>" +
              "<a class='btn' data-wysihtml5-command='spellcheck'> + "
              "<i class='icon-spellchecker spellchecker-button-icon'></i></a>" +
          "</li>"
    },
    stylesheets: [
      "./css/jquery.spellchecker.css"
    ],     
  });

  var wysihtml5 = element.data('wysihtml5');
  
  var body = $(wysihtml5.editor.composer.iframe).contents().find('body');

  var spellchecker = new $.SpellChecker(body, {
    lang: 'en',
    parser: 'html',
    engine: {
      path: "../php/spellchecker.php",
      driver: 'pspell'
    },
    suggestBox: {
      position: 'below'
    }
  });

  // Bind spellchecker handler functions
  spellchecker.on('check.success', function() {
    alert('There are no incorrectly spelt words.');
  });

  wysihtml5.toolbar.find('[data-wysihtml5-command="spellcheck"]').click(function() {
    spellchecker.check();
  });

})();
