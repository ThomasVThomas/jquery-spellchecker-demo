HTML parser usage

// Init the html spellchecker
var spellchecker = new $.SpellChecker('#text-content', {
  parser: 'html',
  engine: {
    path: '../php/spellchecker.php',
    driver: {
      lang: 'en',
      type: 'PSpell',
      dictionary: ''
    }
  },
  suggestBox: {
    position: 'below'
  }
});

// Bind spellchecker handler functions
spellchecker.on('check.success', function() {
  alert('There are no incorrectly spelt words!');
});

// Check the spelling when user clicks on button
$("#check-spelling-button").click(function(e){
  spellchecker.check();
});

Text parser usage

// Init the text spellchecker
var spellchecker = new $.SpellChecker(element, {
  parser: 'text',
  engine: {
    path: '../php/spellchecker.php',
    driver: {
      lang: 'en',
      type: 'google'
    }
  },
  suggestBox: {
    position: 'above'
  },
  incorrectWords: {
    container: '#incorrect-word-list'
  }
});

// Bind spellchecker handler functions
spellchecker.on('check.success', function() {
  alert('There are no incorrectly spelt words.');
});

// Check the spelling when user clicks on button
$("#check-spelling-button").click(function(e){
  spellchecker.check();
});
